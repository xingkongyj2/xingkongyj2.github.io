import jieba
import wordcloud
from scipy.misc import imread
mask = imread("timg2.jpg")
excludes = { }
f = open("aa.txt", "r", encoding="utf-8")
t = f.read()
f.close()
ls = jieba.lcut(t)
txt = " ".join(ls)
w = wordcloud.WordCloud(\
    width = 1000, height = 700,\
    background_color = "white", mask = mask)
w.generate(txt)
w.to_file("grwordcloudm.txt")
