计算函数运行时间

2017-03-30
算法

～～～～～～

#include <stdio.h>
#include <time.h>
#include <math.h>
clock_t start,stop;               //clock_t   是函数clock（）返回数的数据类型 
double duration;
int main()
{
    start=clock();                 
    for(int i=0;i<1000000000;i++)          //重复调用函数，时间增多 
        pow(123,3);
    stop=clock();
    duration=(double)(stop-start)/CLOCKS_PER_SEC;        //转化为时间Sec 
    printf("%f\n",duration);
    printf("%d\n",CLK_TCK);              //每台机器的CLOCKS_PER_SEC的值是不一样的，我的这台是1000 
    return 0;
}
